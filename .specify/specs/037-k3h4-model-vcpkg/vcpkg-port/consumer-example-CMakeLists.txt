# consumer-example-CMakeLists.txt
# Example CMakeLists.txt using banana-k3h4-model (model-only port)

cmake_minimum_required(VERSION 3.20)
project(k3h4-consumer-example)

# Find the banana-k3h4-model package
find_package(banana-k3h4-model REQUIRED CONFIG)

# (Optional) Find native ABI if you need clustering queries
# find_package(banana-native-abi REQUIRED CONFIG)

# Create an executable
add_executable(k3h4_loader consumer-example.cpp)

# Link the model target (this is an INTERFACE library, not a link library)
# It mainly exposes the INTERFACE_PAYLOAD_LOCATION property
target_link_libraries(k3h4_loader PRIVATE banana::k3h4::model)

# Get the payload location for use in code
get_target_property(K3H4_PAYLOAD banana::k3h4::model INTERFACE_MODEL_PAYLOAD_LOCATION)
target_compile_definitions(k3h4_loader PRIVATE 
    K3H4_MODEL_PAYLOAD_PATH="${K3H4_PAYLOAD}"
)

# Alternative: Pass as environment variable at runtime
# (The consumer-example.cpp code checks std::getenv("K3H4_MODEL_PAYLOAD_PATH"))

# Optional: If using native ABI, link it too
# target_link_libraries(k3h4_loader PRIVATE banana::native::library)

# Optional: Print diagnostic info
if(CMAKE_VERBOSE_MAKEFILE)
    message(STATUS "K3H4 Model Payload: ${K3H4_PAYLOAD}")
    message(STATUS "K3H4 Model Manifest: ${banana-k3h4-model_MANIFEST_PATH}")
    message(STATUS "K3H4 Bundle Version: ${banana-k3h4-model_BUNDLE_VERSION}")
endif()
